\subsection{Eccezioni}
\subsubsection*{Dativo e ablativo plurale in -ubus}
\begin{itemize}
    \item Bisillabi in -cus (e.g. lacus)
    \item \textit{Artus, Partus, Tribus}
    \item \textit{Portus, Veru} (Hanno anche l'uscita normale in -\textit{ibus})
\end{itemize}
